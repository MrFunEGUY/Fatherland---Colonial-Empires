fatherland_iron_curtain_rogue_effect = {

	random_list = {
		40 = {
			capital_scope = { weighted_random_owned_pop_group = { kill_pop_group = { pop_group = this amount = 10% } } }

				effect = {
					set_owner = prev
					
					create_ship = {
						name = random
						random_existing_design = colonizer
					}
					set_location = prev.capital_scope
				}
			}
		}

		limit = { has_country_flag = fatherland_roguesetup_explorers }

	}
	else_if = {
		limit = { has_country_flag = fatherland_roguesetup_biotrophy }
		remove_country_flag = fatherland_roguesetup_biotrophy
	}
	else_if = {
		limit = { has_country_flag = fatherland_roguesetup_corpo }
		remove_country_flag = fatherland_roguesetup_corpo
	}
	else_if = {
		limit = { has_country_flag = fatherland_roguesetup_family_corpo }
		remove_country_flag = fatherland_roguesetup_family_corpo
	}
	else_if = {
		limit = { has_country_flag = fatherland_roguesetup_crusaders }
		remove_country_flag = fatherland_roguesetup_crusaders
	}
	else_if = {
		limit = { has_country_flag = fatherland_roguesetup_dissidents }

	else_if = {

	}

		remove_modifier = rogue_target

		clear_global_event_target = rogue_event_planet
	}
}

get_rogue_colony_after_effect = {
	copy_techs_from = { target = root }
	save_event_target_as = rogue_colony_nation
	set_graphical_culture = event_target:fatherland_empire
	get_fatherland_country_info = yes

	give_technology = { tech = tech_starbase_3 message = no }
	give_technology = { tech = tech_strike_craft_1 message = no }

	if = { 
		limit = { has_global_flag = fatherland_instantcomms }
		establish_communications_no_message = event_target:fatherland_empire
	}

	if = { 
		limit = { has_global_flag = fatherland_bonusresources }
		random_list = {
			50 = { set_variable = { which = rogue_resources value = 1 } }
			30 = { set_variable = { which = rogue_resources value = 2 } }
			15 = { set_variable = { which = rogue_resources value = 3 } }
			5 = { set_variable = { which = rogue_resources value = 4 } }
		}

		add_resource = {
			energy = 1000
			minerals = 1000
			food = 2000
			consumer_goods = 500
			alloys = 500
			influence = 150
			minor_artifacts = 200
			mult = rogue_resources
		}

		if = {
			limit = { has_global_flag = cultural_overhaul_funefork_active }
			add_resource = { ecaf_crew = 500 }
		}

		clear_variable = rogue_resources

		add_500_unity_per_year_passed = yes
	}

	get_fatherland_leader_homeworld = yes
	create_leader = {
		class = random_ruler
		species = owner_main_species
		name = random
		skill = 5
		traits = { trait = random_traits }
		background_planet = event_target:leader_homeworld
	}

	if = {
		limit = { exists = ruler }
		ruler = { kill_leader = { show_notification = no } }
	}
	
	set_leader = last_created_leader
}

get_fatherland_leader_homeworld = {
	event_target:fatherland_empire = {
		random_owned_planet = {
			limit = { colony_age >= 10 }
			save_event_target_as = leader_homeworld
		}
	}
}

get_fatherland_country_info = {
	if = {
		limit = { event_target:fatherland_empire = { has_fatherland_country_flag = no } }
		event_target:global_event_country = {
			if = {
				limit = { NOT = { is_variable_set = fatherland_country_count } }
				set_variable = { which = fatherland_country_count value = 1 }
			}
			else = {
				change_variable = { which = fatherland_country_count value = 1 }
			}
		}
		set_fatherland_country_flags = yes
	}
	else = { event_target:fatherland_empire = { find_fatherland_number_flag = yes } }
	event_target:fatherland_empire = {
		set_country_flag = fatherland_of_@event_target:rogue_colony_nation
	}
}

	if = {
		limit = { has_country_flag = fatherland_country_1_flag }
		if = {
			limit = { exists = event_target:rogue_colony_nation }
			event_target:rogue_colony_nation = {
				set_country_flag = fatherland_country_1_leaver
			}
		}
		else_if = {
			limit = { has_origin = origin_rogue_colony_fl }
			save_global_event_target_as = fatherland_country_1
		}
		else_if = {
			limit = { any_country = { has_country_flag = fatherland_country_1_leaver } }
			set_country_flag = country_needed_for_loc
		}
	}

	else_if = {
