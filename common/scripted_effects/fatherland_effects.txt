get_rogue_colony_after_effect = {
	copy_techs_from = { target = root }
	save_event_target_as = rogue_colony_nation
	set_graphical_culture = event_target:fatherland_empire
	get_fatherland_country_info = yes

	give_technology = { tech = tech_starbase_3 message = no }
	give_technology = { tech = tech_strike_craft_1 message = no }

	if = { 
		limit = { has_global_flag = fatherland_instantcomms }
		establish_communications_no_message = event_target:fatherland_empire
	}

	if = { 
		limit = { has_global_flag = fatherland_bonusresources }
		random_list = {
			50 = { set_variable = { which = rogue_resources value = 1 } }
			30 = { set_variable = { which = rogue_resources value = 2 } }
			15 = { set_variable = { which = rogue_resources value = 3 } }
			5 = { set_variable = { which = rogue_resources value = 4 } }
		}

		add_resource = {
			energy = 1000
			minerals = 1000
			food = 2000
			consumer_goods = 500
			alloys = 500
			influence = 150
			minor_artifacts = 200
			mult = rogue_resources
		}

		if = {
			limit = { has_global_flag = cultural_overhaul_funefork_active }
			add_crew_effect = yes
		}

		clear_variable = rogue_resources

		add_500_unity_per_year_passed = yes
	}

	get_fatherland_leader_homeworld = yes
	create_leader = {
		class = random_ruler
		species = owner_main_species
		name = random
		skill = 5
		background_planet = event_target:leader_homeworld
		effect = {
			add_trait = { trait = random_common }
		}
	}

	if = {
		limit = { exists = ruler }
		ruler = { kill_leader = { show_notification = no } }
	}
	
	set_leader = last_created_leader
}

get_fatherland_leader_homeworld = {
	event_target:fatherland_empire = {
		random_owned_planet = {
			limit = { colony_age >= 10 }
			save_event_target_as = leader_homeworld
		}
	}
}

get_fatherland_country_info = {
	if = {
		limit = { event_target:fatherland_empire = { has_fatherland_country_flag = no } }
		event_target:global_event_country = {
			if = {
				limit = { NOT = { is_variable_set = fatherland_country_count } }
				set_variable = { which = fatherland_country_count value = 1 }
			}
			else = {
				change_variable = { which = fatherland_country_count value = 1 }
			}
		}
		set_fatherland_country_flags = yes
	}
	else = { event_target:fatherland_empire = { find_fatherland_number_flag = yes } }
	event_target:fatherland_empire = {
		set_country_flag = fatherland_of_@event_target:rogue_colony_nation
	}
}
