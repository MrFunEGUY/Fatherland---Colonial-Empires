namespace = fatherland_fix

# Change Civics to Origins
country_event = {
	id = fatherland_fix.1000
	is_triggered_only = yes	
	hide_window = yes	

	trigger = {
		any_country = {
			OR = {
				has_civic = civic_banished_house
				has_civic = civic_crusader_state
				has_civic = civic_perscuted_minority
				has_civic = civic_ideological_dissidents
				has_civic = civic_plucky_explorers
				has_civic = civic_rogue_ai
				has_civic = civic_free_slaves
				has_civic = civic_free_proles
				has_civic = civic_free_synths
				has_civic = civic_liberated_biotrophies
			}
		}
	}

	immediate = {
		if = {
			limit = { any_country = { has_civic = civic_banished_house } }
			every_country = {
				limit = { has_civic = civic_banished_house }
				force_remove_civic = civic_banished_house
				set_origin = origin_banished_house
			}
		}
		if = {
			limit = { any_country = { has_civic = civic_crusader_state } }
			every_country = {
				limit = { has_civic = civic_crusader_state }
				force_remove_civic = civic_crusader_state
				set_origin = origin_crusader_state
			}
		}
		if = {
			limit = { any_country = { has_civic = civic_perscuted_minority } }
			every_country = {
				limit = { has_civic = civic_perscuted_minority }
				force_remove_civic = civic_perscuted_minority
				set_origin = origin_perscuted_minority
			}
		}
		if = {
			limit = { any_country = { has_civic = civic_ideological_dissidents } }
			every_country = {
				limit = { has_civic = civic_ideological_dissidents }
				force_remove_civic = civic_ideological_dissidents
				set_origin = origin_ideological_dissidents
			}
		}
		if = {
			limit = { any_country = { has_civic = civic_plucky_explorers } }
			every_country = {
				limit = { has_civic = civic_plucky_explorers }
				force_remove_civic = civic_plucky_explorers
				set_origin = origin_plucky_explorers
			}
		}
		if = {
			limit = { any_country = { has_civic = civic_rogue_ai } }
			every_country = {
				limit = { has_civic = civic_rogue_ai }
				force_remove_civic = civic_rogue_ai
				set_origin = origin_rogue_ai
			}
		}
		if = {
			limit = { any_country = { has_civic = civic_free_slaves } }
			every_country = {
				limit = { has_civic = civic_free_slaves }
				force_remove_civic = civic_free_slaves
				set_origin = origin_free_slaves
			}
		}
		if = {
			limit = { any_country = { has_civic = civic_free_proles } }
			every_country = {
				limit = { has_civic = civic_free_proles }
				force_remove_civic = civic_free_proles
				set_origin = origin_free_proles
			}
		}
		if = {
			limit = { any_country = { has_civic = civic_free_synths } }
			every_country = {
				limit = { has_civic = civic_free_synths }
				force_remove_civic = civic_free_synths
				set_origin = origin_free_synths
			}
		}
		if = {
			limit = { any_country = { has_civic = civic_liberated_biotrophies } }
			every_country = {
				limit = { has_civic = civic_liberated_biotrophies }
				force_remove_civic = civic_liberated_biotrophies
				set_origin = origin_liberated_biotrophies
			}
		}
	}	
}
